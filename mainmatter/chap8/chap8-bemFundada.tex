\section{RelaĂ§Ăµes bem fundadas}

Uma relaĂ§ĂŁo definida Ă© um sĂ­mbolo de predicado binĂˇrio da linguagem em uso, seja ele primitivo, como $\in$ e $=$, seja ele inserido por definiĂ§ĂŁo.

\begin{definition}[$\BST$]
    Uma relaĂ§ĂŁo definida $R$ Ă© dita \emph{set-like} se para todo $x$ existe um (Ăşnico) conjunto $y$ tal que, para todo $z$, $R(z,x)$ se, e somente se, $z\in y$.

    Ou seja, se $\pred_{R}(x)=\{z: R(z,x)\}$ Ă© um conjunto para todo $x$.

    $R$ Ă© dita bem fundada se:

    \[
        \forall X\big(X\neq \emptyset \rightarrow \exists x\in X \, \forall y\big(y\in X \rightarrow \neg R(y,x)\big)\big)
    \]

    Ou seja, $R$ Ă© bem fundada se todo conjunto nĂŁo vazio possui um elemento $R$-minimal.
\end{definition}

A definiĂ§ĂŁo acima Ă©, em realidade, um esquema de definiĂ§Ăµes.
Para cada possĂ­vel sĂ­mbolo de predicado binĂˇrio $R$, temos uma definiĂ§ĂŁo de $R$ ser set-like e uma definiĂ§ĂŁo de $R$ ser bem fundada.


\begin{example}[$\BST$]A relaĂ§ĂŁo $\in$ Ă© set-like, pois para todo $x$, $\pred_{\in}(x)=\{z: z\in x\}=x$ Ă© um conjunto.
\end{example}

\begin{example}[$\BST$]
    A relaĂ§ĂŁo $\preceq$, ``injeta em'', nĂŁo Ă© set-like, pois, para todo $x$, temos $\{x\}\preceq 1$.
    Desse modo, se $\pred_{\preceq}(1)$ fosse um conjunto, ele conteria todos os singletos, o que Ă© impossĂ­vel conforme visto no Lema~\ref{lem:naoExisteConjuntoDeSingletos}.
\end{example}

\begin{example}[$\BST$]
    A relaĂ§ĂŁo $\subseteq$ Ă© set-like se, e somente se, vale o Axioma da PotĂŞncia.
    Nesse caso, $\pred_{\subseteq}(x)=\mathcal{P}(x)$.
\end{example}

\begin{example}[$\BST$]
    A relaĂ§ĂŁo $=$ Ă© set-like, pois $\pred_{=}(x)=\{x\}$ Ă© um conjunto para todo $x$.
\end{example}
\begin{definition}[$\BST$]
    Seja $R$ uma relaĂ§ĂŁo definida.

    Para cada nĂşmero natural $n\geq 2$ e cada $a, b$, um $R$-caminho de comprimento $n$ de $a$ atĂ© $b$ Ă© uma sequĂŞncia de elementos $(x_0, \dots, x_{n-1})$ tal que $x_i R x_{i+1}$ para todo $i$ com $i<n-1$, em que $x_0=a$ e $x_{n-1}=b$.

    Define-se a relaĂ§ĂŁo $R^*$ por $x R^* y$ se, e somente se, existe um $R$-caminho de comprimento $n$, para algum nĂşmero natural $n\geq 2$, que comeĂ§a em $x$ e termina em $y$.

    $R^*$ Ă© o \emph{fecho transitivo} de $R$, ou a \emph{transitivizaĂ§ĂŁo} de $R$.
\end{definition}

ObservaĂ§ĂŁo: a definiĂ§ĂŁo acima estĂˇ marcada como factĂ­vel em $\BST$ apesar de usar a noĂ§ĂŁo de nĂşmero natural pois Ă© possĂ­vel definir nĂşmeros naturais mesmo sem estar de posse do Axioma do Infinito (ver ExercĂ­cio~\ref{exr:caracterizacaoDeNaturaisComoOrdinais}).

As proposiĂ§Ăµes abaixo sĂŁo esquemas de proposiĂ§Ăµes.
Ou seja, para cada possĂ­vel sĂ­mbolo de predicado binĂˇrio $R$ satisfazendo as hipĂłteses, temos uma proposiĂ§ĂŁo.

\begin{proposition}[$\BST$]
    Se $R$ Ă© uma relaĂ§ĂŁo definida, entĂŁo $R^*$ Ă© uma relaĂ§ĂŁo transitiva que estende $R$.
\end{proposition}
\begin{proof}
    Se $x R y$, entĂŁo $x R^* y$, pois $(x,y)$ Ă© um $R$-caminho de comprimento $2$ de $x$ atĂ© $y$.

    Se $x R^* y$ e $y R^* z$, entĂŁo existem $R$-caminhos de comprimento $n$ de $x$ atĂ© $y$, digamos, $(x_0, \dots, x_{n-1})$, e um $R$-caminho de comprimento $m$ de $y$ atĂ© $z$, digamos, $(y_0, \dots, y_{m-1})$, em que $n, m\geq 2$.
    EntĂŁo $(x_0, \dots, x_{n-1}, y_1, \dots, y_{m-1})$ Ă© um $R$-caminho de comprimento $n+m-1$ de $x$ atĂ© $z$, e, portanto, $x R^* z$.
\end{proof}

Na verdade, dada uma relaĂ§ĂŁo definida $R$, o fecho transitivo de $R$ Ă© a menor relaĂ§ĂŁo transitiva que estende $R$ (ver ExercĂ­cio~\ref{ex:fechoTransitivo}).

\begin{proposition}[$\ZFmP$]
    Se $R$ Ă© uma relaĂ§ĂŁo definida set-like, entĂŁo $R^*$ Ă© uma relaĂ§ĂŁo definida set-like.
\end{proposition}

\begin{proof}
    Por induĂ§ĂŁo em $n$, vejamos que para todo $n$ natural, existe um (Ăşnico) conjunto $y$ tal que, para todo $z$, $R^*(z,x)$ se, e somente se, existe um $R$-caminho de comprimento $n\geq 2$ de $z$ atĂ© $x$, que chamaremos de $\pred^n_{R}(x)$.

    Para $n=0$ ou $n=1$, note que $\pred^n_{R}(x)=\emptyset$.

    Para $n=2$, note que $\pred^2_{R}(x)=\pred_{R}(x)$.

    Suponha que o resultado seja verdadeiro para $n\geq 2$.
    Provaremos para $n+1$.

    Por hipĂłtese de induĂ§ĂŁo, existe o conjunto $\pred^n_{R}(x)$ cujos elementos sĂŁo exatamente os $z$ tais que existe um $R$-caminho de comprimento $n$ de $z$ atĂ© $x$.
    Seja $\pred^{n+1}_{R}(x)=\bigcup\{\pred_{R}(y): y\in \pred^n_{R}(x)\}$.

    Se $z\in \pred^{n+1}_{R}(x)$, entĂŁo $z\in \pred_{R}(y)$ para algum $y\in \pred^n_{R}(x)$, e, portanto, existe um $R$-caminho de comprimento $n$ de $y$ atĂ© $x$, digamos, $(x_0, \dots, x_{n-1})$.
    Considere $(z, x_0, \dots, x_{n-1})$.

    Reciprocamente, seja $(x_0, x_1, \dots, x_n)$ um $R$-caminho de comprimento $n+1$ de $x_0$ atĂ© $x_n=x$.
    EntĂŁo $(x_1, \dots, x_n)$ Ă© um $R$-caminho de comprimento $n$ de $x_1$ atĂ© $x$, e, portanto, $x_1\in \pred^n_{R}(x)$.
    Como $x_0 R x_1$, $x_0\in \pred_{R}(x_1)$, e, portanto, $x_0\in \pred^{n+1}_{R}(x)$.

    Considere $Z=\bigcup_{n\in \omega} \pred^n_{R}(x)$.
    Temos que, para todo $z$, $R^*(z,x)$ se, e somente se, $z\in Z$.
\end{proof}

\begin{proposition}[$\ZFmP$](PrincĂ­pio da minimalidade)\label{prop:principioDaMinimalidade}
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada.
    
    Seja $\phi(x, t_1, \dots, t_n)$ uma fĂłrmula.
    EntĂŁo:
    \[
        \forall t_1\dots, t_n\bigg(\exists x \, \phi(x, t_1, \dots, t_n) \rightarrow \exists x\bigg(\phi(x, t_1, \dots, t_n)\wedge \forall y\big(y R x \rightarrow \neg \phi(y, t_1, \dots, t_n)\big)\bigg)\bigg)
    \]
\end{proposition}

\begin{proof}
    Fixe $t_1, \dots, t_n$ e suponha que $\exists x\, \phi(x, t_1, \dots, t_n)$.
    Fixe $x$ dessa forma.

    Se $\forall y\big(y R x \rightarrow \neg \phi(y, t_1, \dots, t_n)\big)$, segue a tese.
    Se nĂŁo, considere $X=\{z \in \pred_{R^*}(x): \phi(z, t_1, \dots, t_n)\}$.
    Temos que $X$ Ă© um conjunto e $X\neq \emptyset$.
    Logo, existe um elemento $R$-minimal de $X$, digamos, $x'$.
    Temos que $\phi(x', t_1, \dots, t_n)$ e, dado $y$ tal que $y R x'R^*x$, segue que $y\in \pred_{R^*}(x)$, e, portanto, $y\notin X$, ou seja, $\neg \phi(y, t_1, \dots, t_n)$.
\end{proof}

\begin{corollary}[$\ZFmP$](PrincĂ­pio da induĂ§ĂŁo)\label{cor:principioDaInducao}
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada.
    
    Seja $\phi(x, t_1, \dots, t_n)$ uma fĂłrmula.
    EntĂŁo:
    \[
        \forall t_1\dots, t_n\bigg(\forall x \, \left(\forall z \in \pred_{R}(x) \, \phi(z, t_1, \dots, t_n) \rightarrow \, \phi(x, t_1, \dots, t_n)\right)\rightarrow \forall x\, \phi(x, t_1, \dots, t_n)\bigg)
    \]
\end{corollary}

\begin{proof}
    Fixe $t_1, \dots, t_n$ e suponha que existe $x$ tal que $\neg \phi(x, t_1, \dots, t_n)$.

    Pela ProposiĂ§ĂŁo~\ref{prop:principioDaMinimalidade}, existe um elemento $R$-minimal $x'$ tal que $\neg \phi(x', t_1, \dots, t_n)$ e $\phi(z, t_1, \dots, t_n)$ para todo $z\in \pred_{R}(x')$.

    Por hipĂłtese, temos que $\forall x \, \left(\forall z \in \pred_{R}(x) \, \phi(z, t_1, \dots, t_n) \rightarrow \, \phi(x, t_1, \dots, t_n)\right)$.
    Aplicando isso para $x'$, temos que $\phi(x', t_1, \dots, t_n)$, o que Ă© uma contradiĂ§ĂŁo.
\end{proof}


\begin{theorem}[$\ZFmPI$](Esquema da recursĂŁo transfinita)\label{thm:esquemaDaRecursaoTransfinita}\index{recursĂŁo transfinita}
    Suponha que um sĂ­mbolo de funĂ§ĂŁo binĂˇrio $G(p, s, a)$ seja introduzido por definiĂ§ĂŁo.

    EntĂŁo Ă© possĂ­vel introduzir um sĂ­mbolo de funĂ§ĂŁo binĂˇrio $F(p, a)$ de modo que
    \[
        \forall a \forall p \Big( F_{p}(a)=G(p, F_p|\pred_{R^*}(a), a)\Big).
    \]
\end{theorem}
Para provar esse teorema, faĂ§amos as seguintes definiĂ§Ăµes auxiliares.

\begin{definition}[$\ZFmPI$]
    Seja $R$ uma relaĂ§ĂŁo definida set-like.
    Dizemos que um conjunto $E$ Ă© $R$-fechado se, e somente se, para todo $x\in E$, $\pred_{R}(x)\subseteq E$.

    Dados conjuntos $t, a, p$, escrevemos $\Ap_G(t, p, a)$ e dizemos que \emph{$t$ Ă© uma computaĂ§ĂŁo parcial de domĂ­nio $E$ em $a$ com parĂ˘metro $p$ baseada em $G$} se, e somente se,
    \begin{itemize}
        \item $t$ Ă© uma funĂ§ĂŁo,
        \item $\dom(t)$ Ă© $R$-fechado,
        \item $a \in \dom(t)$,
        \item $G(p, t|\pred_{R}(b), b)$ Ă© definido para todo $b\in \dom(t)$, e
        \item $G(p, t|\pred_{R}(b), b)=t(b)$ para todo $b\in \dom(t)$.
    \end{itemize}
\end{definition}

\begin{lemma}[$\ZFmPI$]
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada e $G$ um sĂ­mbolo de funĂ§ĂŁo binĂˇrio introduzido por definiĂ§ĂŁo.

    Se $t_1$, $t_2$, $a$, $p$ sĂŁo tais que $\Ap_G(t_1, p, a)$ e $\Ap_G(t_2, p, a)$, entĂŁo $t_1$ e $t_2$ sĂŁo compatĂ­veis.
\end{lemma}
\begin{proof}
    Suponha por absurdo que nĂŁo.
    EntĂŁo seja $x \in \dom(t_1)\cap \dom(t_2)$ $R$-minimal tal que $t_1(x)\neq t_2(x)$.

    Dessa forma, para todo $yRx$, temos que $y\in \dom(t_1)\cap \dom(t_2)$, pois $\dom(t_1)$ e $\dom(t_2)$ sĂŁo $R$-fechados, e, portanto, $t_1(y)=t_2(y)$ pela minimalidade de $x$.
    DaĂ­, segue que:
    \[
        t_1(x)=G(p, t_1|\pred_{R}(x), x)=G(p, t_2|\pred_{R}(x), x)=t_2(x),
    \]
    o que Ă© uma contradiĂ§ĂŁo.
\end{proof}

\begin{lemma}
    Seja $R$ uma relaĂ§ĂŁo definida set-like.

    Seja $a$ um conjunto.
    EntĂŁo $\pred_{R^*}(a)\cup\{a\}$ Ă© $R$-fechado.
    AlĂ©m disso, se $E$ Ă© um conjunto $R$-fechado e $a\in E$, entĂŁo $\pred_{R^*}(a)\cup\{a\}\subseteq E$. 
\end{lemma}
\begin{proof}
    Para a primeira afirmaĂ§ĂŁo, seja $x\in \pred_{R^*}(a)\cup\{a\}$.
    Devemos ver que $\pred_{R}(x)\subseteq \pred_{R^*}(a)\cup\{a\}$.
    Se $x=a$, entĂŁo $\pred_{R}(x)\subseteq \pred_{R^*}(a)\cup\{a\}$, pois $\pred_{R}(x)\subseteq \pred_{R^*}(a)$.
    Se $x\in \pred_{R^*}(a)$, tome $y\in \pred_{R}(x)$.
    Temos que $yRxR^*a$, e, portanto, $y\in \pred_{R^*}(a)$.

    Agora vamos provar a segunda afirmaĂ§ĂŁo.
    Dado $b\in \pred_{R^*}(a)$, existe um $R$-caminho de comprimento $n$ de $b$ atĂ© $a$, digamos, $(x_0, \dots, x_{n-1})$, em que $x_0=b$ e $x_{n-1}=a$.
    Se $b \notin E$, existe o maior $i<n$ tal que $x_i \notin E$.
    Temos que $i<n-1$, pois $a=x_{n-1}\in E$.
    Assim, $x_{i+1}\in E$ e $x_i R x_{i+1}$, o que Ă© uma contradiĂ§ĂŁo.
\end{proof}

\begin{corollary}
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada e $G$ um sĂ­mbolo de funĂ§ĂŁo binĂˇrio introduzido por definiĂ§ĂŁo.

    Sejam $a$ e $p$ conjuntos.
    Se existe $t$ tal que $\Ap_G(t, p, a)$, entĂŁo existe um Ăşnico $t'$ tal que $\Ap_G(t', p, a)$ e $\dom(t')=\pred_{R^*}(a)\cup\{a\}$.
\end{corollary}
\begin{proof}
    Para a existĂŞncia, fixado $t$, tome $t'=t|\pred_{R^*}(a)\cup\{a\}$.

    Para a unicidade, se $t'$ e $t''$ sĂŁo tais que $\Ap_G(t', p, a)$ e $\Ap_G(t'', p, a)$ e $\dom(t')=\dom(t'')=\pred_{R^*}(a)\cup\{a\}$, entĂŁo $t'$ e $t''$ sĂŁo compatĂ­veis, e, portanto, $t'=t''$.
\end{proof}
\begin{lemma}
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada e $G$ um sĂ­mbolo de funĂ§ĂŁo binĂˇrio introduzido por definiĂ§ĂŁo.

    Para todo $a$ e $p$, existe um $t$ tal que $\Ap_G(t, p, a)$.
\end{lemma}

\begin{proof}
    Fixe $p$.
    Provaremos por $R$-induĂ§ĂŁo que, para todo $a$, existe um $t$ tal que $\Ap_G(t, p, a)$.

    Dado $a$, basta ver que se a tese vale para todo $b\in \pred_{R}(a)$, entĂŁo vale para $a$.

    Suponha que, para todo $c\in \pred_{R}(a)$, existe um $t$ tal que $\Ap_G(t, p, c)$.
    Seja $t_c$ o Ăşnico $t$ tal que $\Ap_G(t, p, c)$ e $\dom(t)=\pred_{R^*}(c)\cup\{c\}$.
    Seja $t'=\bigcup_{c\in \pred_{R}(a)} t_c$ e $t=t'\cup\{(a, G(p, t'|\pred_{R}(a), a))\}$.

    Temos que $\dom t=\bigcup_{c\in \pred_{R}(a)} \dom t_c \cup \{a\}=\bigcup_{c\in \pred_{R}(a)} (\pred_{R^*}(c)\cup\{c\})\cup \{a\}=\pred_{R^*}(a)\cup\{a\}$, Ă© fechado e $a\in \dom t$.

    Dado $b\in \dom t$, se $b=a$, entĂŁo $t(a)=G(p, t|\pred_{R}(a), a)=G(p, t'|\pred_{R}(a), a)$.

    JĂˇ se $b\neq a$, existe um $c\in \pred_{R}(a)$ tal que $b\in \pred_{R^*}(c)\cup\{c\}$, e, portanto, $\pred_{R}(b)\subseteq \pred_{R^*}(c)\cup\{c\}\subseteq \dom t_c$.
    Assim, $t|\pred_{R}(b)=t_c|\pred_{R}(b)$, e, portanto, $t(b)=G(p, t|\pred_{R}(b), b)=G(p, t_c|\pred_{R}(b), b)=t_c(b)$.
\end{proof}

Abaixo, re-enunciamos o Teorema~\ref{thm:esquemaDaRecursaoTransfinita} de maneira mais precisa.

\begin{theorem}[$\ZFmPI$](Esquema da recursĂŁo transfinita)
    Suponha que um sĂ­mbolo de funĂ§ĂŁo binĂˇrio $G(p, s, a)$ seja introduzido por definiĂ§ĂŁo.

    Seja $F^G$ o sĂ­mbolo de funĂ§ĂŁo binĂˇrio introduzido por definiĂ§ĂŁo tal que:
    \[
        \forall a \forall p \forall y\bigg(F^G_{p}(a)=y \leftrightarrow \exists t\big(\Ap_G(t, p, a)\wedge t(a)=y\big)\bigg).
    \]
    EntĂŁo $\forall a \forall p \Big( F^G_{p}(a)=G(p, F^G_p|\pred_{R^*}(a), a)\Big)$.
\end{theorem}
\begin{proof}
    Primeiro, notemos que $F^G$ Ă© bem definida: dado $a$ e $p$, existe um $t$ tal que $\Ap_G(t, p, a)$, e, se $t'$ Ă© outro tal que $\Ap_G(t', p, a)$, entĂŁo $t'$ e $t$ sĂŁo compatĂ­veis, e, portanto, $t'(a)=t(a)$.

    Segundo, dado $a$ e $p$, seja $t$ tal que $\Ap_G(t, p, a)$.
    EntĂŁo, para todo $b\in \pred_{R}(a)$, temos que $\Ap_G(t, p, b)$, e, portanto, $t|\pred_{R}(a)=F^G_p|\pred_{R}(a)$.
    Assim, $F^G_p(a)=t(a)=G(p, t|\pred_{R}(a), a)=G(p, F^G_p|\pred_{R}(a), a)$.
\end{proof}

Com isso, podemos definir o rank com respeito a uma relaĂ§ĂŁo definida set-like e bem fundada $R$.
\begin{definition}[$\ZFmPI$]
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada.
    O \emph{rank} de um conjunto $x$ com respeito a $R$, denotado por $\rank_R(x)$, Ă© definido recursivamente pela relaĂ§ĂŁo $R$ a partir da seguinte fĂłrmula.

    \[
        \rank_R(x)=\bigcup\{\rank_R(y)+1: y R x\}.
    \]
\end{definition}

A definiĂ§ĂŁo acima pode ser formalizada a partir do teorema da recursĂŁo transfinita, tomando
\[
    G(p, s, a)=\bigcup\{s(b)+1: b\in \dom s \land b R a\}.
\]
O Teorema da RecursĂŁo nos define um sĂ­mbolo funcional binĂˇrio $F(p, a)$ tal que, para todos $p, a$:
\[
    F_p(a)=G(p, F_p|\pred_{R^*}(a), a)=\bigcup\{F_p(b)+1: b R a\}.
\]

Tome $p$ qualquer, digamos, $p=\emptyset$, e seja $\rank_R(a)=F_p(a)$.
EntĂŁo $\rank_R(a)=\bigcup\{\rank_R(b)+1: b R a\}$, como querĂ­amos.

Tal funĂ§ĂŁo Ă©, em algum sentido, a Ăşnica funĂ§ĂŁo que satisfaz essa fĂłrmula (ver ExercĂ­cio~\ref{ex:rankBemDefinido}).

\begin{lemma}[$\ZFmP$]
    Seja $R$ uma relaĂ§ĂŁo definida set-like.
    \begin{enumerate}[label=(\roman*)]
        \item Para todo $x$, $\rank_R(x)$ Ă© um ordinal.
        \item Para todo $x$, $\rank_R(x)=\sup\{\rank_R(y)+1: y R x\}$.
        \item Para todos $x$ e $y$, se $y R x$, entĂŁo $\rank_R(y)<\rank_R(x)$.
        \item Para todos $x$ e $y$, se $y R^* x$, entĂŁo $\rank_R(y)<\rank_R(x)$.
    \end{enumerate}
\end{lemma}
\begin{proof}
    Para (i), por $R$-induĂ§ĂŁo, para todo $x$, $\rank_R(x)$ Ă© um ordinal.
    Dado $x$, suponha que, para todo $y R x$, $\rank_R(y)$ Ă© um ordinal.
    EntĂŁo $\rank_R(x)=\bigcup\{\rank_R(y)+1: y R x\}$ Ă© um ordinal.

    Para (ii), de (i), temos que $\rank_R(x)=\bigcup\{\rank_R(y)+1: y R x\}=\sup\{\rank_R(y)+1: y R x\}$, pois o supremo de um conjunto de ordinais Ă© sua uniĂŁo.

    Para (iii), se $yRx$, temos que $\rank_R(y)+1\leq \rank_R(x)$ pela definiĂ§ĂŁo de $\rank_R(x)$, e, portanto, $\rank_R(y)<\rank_R(x)$.

    Para (iv), suponha que $y R^* x$.
    EntĂŁo existe $n\geq 2$ e um $R$-caminho de comprimento $n$ de $y$ atĂ© $x$, digamos, $(x_0, \dots, x_{n-1})$, em que $x_0=y$ e $x_{n-1}=x$.
    Por (iii), $\rank_R(x_0)<\rank_R(x_1)<\cdots <\rank_R(x_{n-1})$, e, portanto, $\rank_R(y)=\rank_R(x_0)<\rank_R(x_{n-1})=\rank_R(x)$.
\end{proof}

\begin{corollary}[$\ZFmP$]
    Se $R$ Ă© uma relaĂ§ĂŁo definida set-like e bem fundada, entĂŁo $R^*$ Ă© set-like e bem fundada.
\end{corollary}
\begin{proof}
    JĂˇ vimos que $R^*$ Ă© set-like.

    Para ver que $R^*$ Ă© bem fundada, seja $X$ um conjunto nĂŁo vazio.
    Seja $O=\{\rank_R(x): x\in X\}$.
    Como $O$ Ă© um conjunto de ordinais, existe um ordinal $\alpha$ tal que $\alpha=\min O$.
    Seja $x\in X$ tal que $\rank_R(x)=\alpha$.
    Se $y R^* x$, entĂŁo $\rank_R(y)<\rank_R(x)=\alpha=\min O$, e, portanto, $\rank_R(y)\notin O$.
    DaĂ­, segue que $y\notin X$, e, portanto, $x$ Ă© um elemento $R^*$-minimal de $X$.
\end{proof}
