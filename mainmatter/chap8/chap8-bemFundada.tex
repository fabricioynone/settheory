\section{RelaĂ§Ăľes bem fundadas}

Uma relaĂ§ĂŁo definida ĂŠ um sĂ­mbolo de predicado binĂĄrio introduzido por definiĂ§ĂŁo.

\begin{definition}[$\BST$]
    Uma relaĂ§ĂŁo definida $R$ ĂŠ dita \emph{set-like} se para todo $x$ existe um (Ăşnico) conjunto $y$ tal que, para todo $z$, $R(z,x)$ se, e somente se, $z\in y$.

    Ou seja, se $y_\downarrow=\pred_{R}(x)=\{z: R(z,x)\}$ ĂŠ um conjunto para todo $x$.

    $R$ ĂŠ dita bem fundada em $A$ se:

    \[
        \forall X\big(X\neq \emptyset \rightarrow \exists x\in X \, \forall y\big(y\in X \rightarrow \neg R(y,x)\big)\big)
    \]

    Ou seja, $R$ ĂŠ bem fundada se todo subconjunto nĂŁo vazio possui um elemento $R$-minimal.
\end{definition}

A definiĂ§ĂŁo acima ĂŠ, em realidade, um esquema de definiĂ§Ăľes.
Para cada possĂ­vel sĂ­mbolo de predicado binĂĄrio $R$, temos uma definiĂ§ĂŁo de $R$ ser set-like e uma definiĂ§ĂŁo de $R$ ser bem fundada.
O mesmo se aplica para a definiĂ§ĂŁo abaixo.

\begin{definition}[$\BST$]
    Seja $R$ uma relaĂ§ĂŁo definida.

    Para cada nĂşmero natural $n\geq 2$ e cada $a, b$, um $R$-caminho de comprimento $n$ de $a$ atĂŠ $b$ ĂŠ uma sequĂŞncia de elementos $(x_0, \dots, x_{n-1})$ tal que $x_i R x_{i+1}$ para todo $i$ com $i<n-1$, em que $x_0=a$ e $x_{n-1}=b$.

    Define-se a relaĂ§ĂŁo $R^*$ por $x R^* y$ se, e somente se, existe um $R$-caminho de comprimento $n$ para algum nĂşmero natural $n$ que comeĂ§a em $x$ e termina em $y$.
\end{definition}

\begin{proposition}[$\BST$]
    Se $R$ ĂŠ uma relaĂ§ĂŁo definida set-like, entĂŁo $R^*$ ĂŠ uma relaĂ§ĂŁo transitiva que estende $R$.
\end{proposition}
\begin{proof}
    Se $x R y$, entĂŁo $x R^* y$ pois $(x,y)$ ĂŠ um $R$-caminho de comprimento $2$ de $x$ atĂŠ $y$.

    Se $x R^* y$ e $y R^* z$, entĂŁo existem $R$-caminhos de comprimento $n$ de $x$ atĂŠ $y$, digamos, $(x_0, \dots, x_{n-1})$, e um $R$-caminho de comprimento $m$ de $y$ atĂŠ $z$, digamos, $(y_0, \dots, y_{m-1})$, em que $n, m\geq 2$.
    EntĂŁo $(x_0, \dots, x_{n-1}, y_1, \dots, y_{m-1})$ ĂŠ um $R$-caminho de comprimento $n+m-1$ de $x$ atĂŠ $z$, e, portanto, $x R^* z$.
\end{proof}

\begin{proposition}[$\ZFmP$]
    Se $R$ ĂŠ uma relaĂ§ĂŁo definida set-like, entĂŁo $R^*$ ĂŠ uma relaĂ§ĂŁo definida set-like.
\end{proposition}

\begin{proof}
    Por induĂ§ĂŁo em $n$, vejamos que para todo $n\in \omega$, existe um (Ăşnico) conjunto $y$ tal que, para todo $z$, $R^*(z,x)$ se, e somente se, existe um $R$-caminho de comprimento $n$ de $z$ atĂŠ $x$, que chamaremos de $\pred^n_{R}(x)$.

    Para $n=0$ ou $n=1$, note que $\pred^0_{R}(x)=\emptyset$.

    Para $n=2$, note que $\pred^1_{R}(x)=\pred_{R}(x)$.

    Suponha que o resultado seja verdadeiro para $n\geq 2$.
    Provaremos para $n+1$.

    Por hipĂłtese de induĂ§ĂŁo, existe o conjunto $\pred^n_{R}(x)$ cujos elementos sĂŁo exatamente os $z$ tais que existe um $R$-caminho de comprimento $n$ de $z$ atĂŠ $x$.
    Seja $\pred^{n+1}_{R}(x)=\bigcup\{\pred_{R}(y): y\in \pred^n_{R}(x)\}$.

    Se $z\in \pred^{n+1}_{R}(x)$, entĂŁo $z\in \pred_{R}(y)$ para algum $y\in \pred^n_{R}(x)$, e, portanto, um $R$-caminho de comprimento $n$ de $y$ atĂŠ $x$, digamos, $(x_0, \dots, x_{n-1})$.
    Considere $(z, x_0, \dots, x_{n-1})$.

    Reciprocamente, seja $(x_0, x_1, \dots, x_n)$ um $R$-caminho de comprimento $n+1$ de $x_0$ atĂŠ $x_n=x$.
    EntĂŁo $(x_1, \dots, x_n)$ ĂŠ um $R$-caminho de comprimento $n$ de $x_1$ atĂŠ $x$, e, portanto, $x_1\in \pred^n_{R}(x)$.
    Como $x_0 R x_1$, $x_0\in \pred_{R}(x_1)$, e, portanto, $x_0\in \pred^{n+1}_{R}(x)$.

    Considere $Z=\bigcup_{n\in \omega} \pred^n_{R}(x)$.
    Temos que, para todo $z$, $R^*(z,x)$ se, e somente se, $z\in Z$.
\end{proof}

\begin{proposition}[$\ZFmP$]
    Seja $R$ uma relaĂ§ĂŁo definida set-like e bem fundada.
    
    Seja $\phi(x, t_1, \dots, t_n)$ uma fĂłrmula.
    EntĂŁo:
    \[
        \forall t_1\dots, t_n\bigg(\exists x\in A \, \phi(x, t_1, \dots, t_n) \rightarrow \exists x\bigg(\phi(x, t_1, \dots, t_n)\wedge \forall y\big(y R x \rightarrow \neg \phi(y, t_1, \dots, t_n)\big)\bigg)\bigg)
    \]
\end{proposition}

\begin{proof}
    Fixe $t_1, \dots, t_n$ e suponha que $\exists x\, \phi(x, t_1, \dots, t_n)$.
    Fixe $x$ dessa forma.

    Se $\forall y\big(y R x \rightarrow \neg \phi(y, t_1, \dots, t_n)\big)$, segue a tese.
    Se nĂŁo, considere $X=\{z \in \pred_{R^*}(x): \phi(z, t_1, \dots, t_n)\}$.
    Temos que $X$ ĂŠ um conjunto e $X\neq \emptyset$.
    Logo, existe um elemento $R$-minimal de $X$, digamos, $x'$.
    Temos que $\phi(x', t_1, \dots, t_n)$ e, dado $y$ tal que $y R x'$, $y\in \pred_{R^*}(x)$, e, portanto, $y\notin X$, ou seja, $\neg \phi(y, t_1, \dots, t_n)$.
\end{proof}

\end{corollary}

\begin{proof}
    Fixe $t_1, \dots, t_n$ e suponha que existe $x$ tal que $\neg \phi(x, t_1, \dots, t_n)$.


\end{proof}
