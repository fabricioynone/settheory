\section{Conjuntos bem fundados}\label{sec:conjuntosBemFundados}

\begin{definition}[$\ZFmP$]
    Seja $x$ um conjunto.
    Definimos $\trcl(x)=\pred_{\in^*}(x)$, onde $\in^*$ é a transitivização de $\in$.
\end{definition}
\begin{definition}[$\ZFmP$]
    Um conjunto $x$ é \textit{bem fundado} se $\in$ é bem fundada em $\trcl(x)$.

    A classe dos conjuntos bem fundados é denotada por $\WF$.
\end{definition}
\begin{lemma}[$\ZFmP$]
    Para quaisquer conjuntos $x$ e $y$:
    \begin{enumerate}[label=(\alph*)]
        \item $\trcl(x)$ é transitivo.
        \item $x\subseteq \trcl(x)$.
        \item Para toda classe transitiva $t$, se $x\subseteq t$, então $\trcl(x)\subseteq t$.
        \item Se $y\subseteq x$, então $\trcl(y)\subseteq \trcl(x)$.
    \end{enumerate}
\end{lemma}
\begin{proof}
    (a) Dado $y\in\trcl(x)$ e $z\in y$, temos que $z\in y\in^* x$.
    Assim, $z\in^* x$, e, portanto, $z\in \trcl(x)$.

    (b) Imediato, pois para todo $y\in x$, $y\in^* x$.

    (c) Seja $t$ uma classe transitiva tal que $x\subseteq t$.
    Por indução em $n\geq 2$, mostraremos que para todo $y$ para o qual existe um $\in$-caminho de $y$ até $x$ de comprimento $n$, $y\in t$.

    Para $n=2$, seja $y$ tal que existe um $\in$-caminho de $y$ até $x$ de comprimento $2$.
    Segue que $y \in x$.
    Como $x\subseteq t$, segue que $y\in t$.

    Suponha que a tese seja verdadeira para algum $n\geq 2$.
    Veremos que também é verdadeira para $n+1$.

    Seja $y$ tal que existe um $\in$-caminho de $y$ até $x$ de comprimento $n+1$.
    Assim, existe $z$ tal que $y\in z$ e existe um $\in$-caminho de $z$ até $x$ de comprimento $n$.
    Por hipótese de indução, $z\in t$.
    Como $t$ é transitivo e $y\in z\in t$, segue que $y\in t$.


    (d) Temos que $y\subseteq x\subseteq \trcl(x)$, sendo este último transitivo.
    Pelo item (c), segue que $\trcl(y)\subseteq \trcl(x)$.
\end{proof}

\begin{corollary}[$\ZFmP$]
    Um conjunto $X$ é bem fundado se, e somente se, $\in$ é bem fundada em alguma classe transitiva $t$ tal que $X\subseteq t$.
\end{corollary}
\begin{proof}
    Se $X$ é bem fundado, então $\in$ é bem fundada em $\trcl(X)$, que é transitiva e contém $X$.

    Já se $\in$ é bem fundada em alguma classe transitiva $t$ que contém $X$, então $\in$ é bem fundada em $\trcl(X)$, pois $\trcl(X)\subseteq t$.
\end{proof}

\begin{lemma}[$\ZFmP$]\label{lem:conjuntosBemFundadosNaoFazemCiclos}
    Se $x$ é um conjunto bem fundado,$x\notin^* x$.
\end{lemma}
\begin{proof}
    Suponha que $x\in^* x$.
    Existe um $\in$-caminho de $x$ até $x$, digamos, $(x_0,\dots,x_n)$, em que $x_0=x$ e $x_n=x$ e $n\geq 2$.

    Temos que para cada $i\in\{0,\dots,n-1\}$, $x_i\in^* x$.
    Assim, $x_i\in \trcl(x)$ para cada $i\leq n$.

    Seja $X=\{x_i: i\in\{0,\dots,n-1\}\}$.
    Então $X\subseteq \trcl(x)$ e $X\neq\emptyset$.
    Seja $x_k$ um elemento $\in_{\trcl(x)}$-minimal de $X$.
    Se $k=0$, temos que $x_0=x_n\ni x_{n-1}$, o que é absurdo.
    Se $k>0$, temos que $x_{k-1}\in x_k$, o que é absurdo.
\end{proof}

\begin{lemma}[$\ZFmP$]
    Seja $x$ um conjunto.
    Então $x$ é bem fundado se, e somente se, $\in$ é bem fundada em $\trcl(x)\cup\{x\}$.
\end{lemma}
\begin{proof}
    ($\Leftarrow$) Imediato, pois $\trcl(x)\subseteq \trcl(x)\cup\{x\}$.

    ($\Rightarrow$) Seja $X\subseteq \trcl(x)\cup\{x\}$ tal que $X\neq\emptyset$.
    Se $x\notin X$, então $X\subseteq \trcl(x)$, e, portanto, existe um elemento $\in_{\trcl(x)}$-minimal de $X$, que é também um elemento $\in_{\trcl(x)\cup\{x\}}$-minimal de $X$ uma vez que $x\notin X$.

    Se $x\in X$, seja $Y=X\setminus\{x\}$.
    Se $Y=\emptyset$, então $x$ é um elemento $\in_{\trcl(x)\cup\{x\}}$-minimal de $X$, pois $x\notin x$ pelo Lema~\ref{lem:conjuntosBemFundadosNaoFazemCiclos}.

    Se $Y\neq\emptyset$, então $Y\subseteq \trcl(x)$, e, portanto, existe um elemento $\in_{\trcl(x)}$-minimal $m$ de $Y$.
    Assim, todo $y\in Y\setminus\{x\}$ é tal que $m\notin y$.
    Devemos ter também que $x\notin m$, do contrário teremos $x\in m\in^* x$, o que é absurdo pelo Lema~\ref{lem:conjuntosBemFundadosNaoFazemCiclos}.
\end{proof}

\begin{definition}[$\ZFmP$]
    Se $x \in \WF$, então $\rank(x)=\rank_{\in, \trcl(x)\cup\{x\}}(x)$.
\end{definition}

\begin{proposition}[$\ZFmP$]
    Se $T$ é uma classe ou conjunto transitivo e bem fundado, então $T\subseteq \WF$ e $\rank_{\in, T}(x)=\rank(x)$ para todo $x\in T$.
\end{proposition}
\begin{proof}
    Para cada $x\in T$, temos que $\trcl(x)\cup\{x\}\subseteq T$, pois $T$ é transitivo.
    Como $T$ é transitiva e bem fundada, todo elemento de $\trcl(x)\cup\{x\}$ é bem fundado, e, portanto, $\trcl(x)\cup\{x\}\subseteq \WF$.
    
    Ademais, como para todo $y\in \trcl(x)\cup\{x\}$ temos que $\pred_{\in_{\trcl(x)\cup\{x\}}^*}(y)=\pred_{\in_T^*}(y)$, segue que $\rank_{\in, T}(x)=\rank_{\in, \trcl(x)\cup\{x\}}(x)=\rank(x)$.
\end{proof}


\begin{proposition}[$\ZFmP$]
    $\WF$ é uma classe transitiva e bem fundada.
    Assim, $\WF$ é a maior classe transitiva e bem fundada.
\end{proposition}
\begin{proof}
    $\WF$ é transitiva, pois, se $x\in \WF$ e $y\in x$, então $y\in \WF$ já que $\in_{\trcl(y)}\subseteq \in_{\trcl(x)}$.

    $\WF$ é bem fundada: dado $X\subseteq \WF$ tal que $X\neq\emptyset$, seja $x\in X$ tal que $\rank(x)$ é mínimo.
    Se $y\in x$, temos que $y\notin X$, pois $\rank(y)=\rank_{\in_{\trcl(y)\cup\{y\}}}(y)=\rank_{\in_{\trcl(x)\cup\{x\}}}(y)<\rank(x)$.
    Assim, $x$ é um elemento $\in$-minimal de $X$.
\end{proof}
\begin{corollary}[$\ZFmP$]
    Se $x\in \WF$, então $\rank(x)=\sup\{\rank(y)+1: y\in x\}$.
\end{corollary}
\begin{proof}
    Seja $x\in \WF$.
    Temos que $\rank(x)=\sup\{\rank(y)+1: y\in_{\trcl(x)\cup\{x\}} y\in x\}=\sup\{\rank(y)+1: y\in x\}$.
\end{proof}
\begin{corollary}[$\ZFmP$]
    Se $x, y \in \WF$ e $x\in^* y$, então $\rank(x)<\rank(y)$.
\end{corollary}
\begin{proof}
    Seja $x, y \in \WF$.
    É suficiente provar que se $x\in y$, então $\rank(x)<\rank(y)$.
    Mas isso é imediato do corolário anterior.
\end{proof}

\begin{corollary}[$\ZFmP$]
    Se $x, y \in \WF$ e $x\subseteq y$, então $\rank(x)\leq \rank(y)$.
\end{corollary}
\begin{proof}
    Temos que $\rank(x)=\sup\{\rank(z)+1: z\in x\}\leq \sup\{\rank(z)+1: z\in y\}=\rank(y)$.
\end{proof}
\begin{corollary}[$\ZFmP$]
    Para $x$ qualquer, temos que $x\in \WF$ se, e somente se, $x\subseteq \WF$.
\end{corollary}
\begin{proof}
    Se $x\in \WF$, então $x\subseteq \WF$ pois $\WF$ é transitiva.

    Se $x\subseteq \WF$, então $\trcl(x)\subseteq \WF$, pois $\WF$ é transitiva.
    Como $\WF$ é bem fundada, então $\in$ é bem fundada em $\trcl(x)$, e, portanto, $x\in \WF$.
\end{proof}

\begin{lemma}[$\ZFmP$]
    Para todo ordinal $\alpha$, $\alpha$ é bem fundado e $\rank(\alpha)=\alpha$.
\end{lemma}
\begin{proof}
    Utilizamos indução transfinita.

    Suponha que para todo $\beta<\alpha$, $\beta \in \WF$ e $\rank(\beta)=\beta$.
    Então $\alpha=\{\beta: \beta<\alpha\}\subseteq \WF$, e, portanto, $\alpha\in \WF$.
    Ademais, $\rank(\alpha)=\sup\{\rank(\beta)+1: \beta\in \alpha\}=\sup\{\beta+1: \beta< \alpha\}=\alpha$.

    Para a última igualdade, com efeito, se $\beta<\alpha$, então $\beta+1\leq \alpha$, logo, $\alpha$ é um majorante de $A=\{\beta+1: \beta< \alpha\}$.
    Esse é o menor majorante, pois se $\xi<\alpha$, então $\xi+1\in A$, e, portanto, $\xi$ não é um majorante de $A$.
\end{proof}

Assim, $\ON\subseteq \WF$.

\begin{corollary}[$\ZFmP$]
    Sejam $x, y\in \WF$.
    Então:
    \begin{enumerate}[label=(\alph*)]
        \item $\rank(\{x, y\})=\max\{\rank(x), \rank(y)\}+1$.
        \item $\rank(\langle x, y\rangle)=\max\{\rank(x), \rank(y)\}+2$.
        \item $\bigcup x\in \WF$ e $\rank(\bigcup x)\leq \rank(x)$.
        \item $x\cup y$ é bem fundado e $\rank(x\cup y)=\max\{\rank(x), \rank(y)\}$.
        \item $\trcl(x)\in \WF$ e $\rank(\trcl(x))=\rank(x)$.
        \item Se existe $\mathcal P(x)$, então $\mathcal P(x)\in \WF$ e $\rank(\mathcal P(x))=\rank(x)+1$.
    \end{enumerate}
\end{corollary}

\begin{proof}
    (a) Temos que $\{x, y\}\subseteq \WF$, logo, $\{x, y\}\in \WF$.

    Ademais, $\rank(\{x, y\})=\sup\{\rank(z)+1: z\in \{x, y\}\}=\max\{\rank(x)+1, \rank(y)+1\}=\max\{\rank(x), \rank(y)\}+1$.

    (b) Como $\langle x, y\rangle=\{\{x\}, \{x, y\}\}$, temos que $\langle x, y\rangle\in \WF$ pelo item (a).
    Ademais, $\rank(\langle x, y\rangle)=\sup\{\rank(z)+1: z\in \langle x, y\rangle\}=\sup\{\rank(\{x\})+1, \rank(\{x, y\})+1\}=\sup\{\rank(x)+2, \max\{\rank(x), \rank(y)\}+2\}=\max\{\rank(x), \rank(y)\}+2$.

    (c)
\end{proof}



\subsection*{Exercícios}
\begin{exercise}[$\ZFmP$]
    Seja $x$ um conjunto.
    Para cada $n$ natural, definimos $\bigcup^n x$ recursivamente de modo que:
    \begin{align*}
        \bigcup^0 x&=x\\
        \bigcup^{n+1} x&=\bigcup(\bigcup^n x).
    \end{align*}    

    Mostre que $\trcl(x)=\bigcup_{n\in\omega}\bigcup^n x$.
\end{exercise}
