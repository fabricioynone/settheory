\section{Conjuntos bem-fundados}\label{sec:conjuntosBemFundados}
Nessa seção, introduziremos a classe dos conjuntos bem-fundados.

Iniciaremos definindo o fecho transitivo de um conjunto.
\begin{definition}[$\ZFmP$]
    Seja $x$ um conjunto.
    O \emph{fecho transitivo} de $x$ é o conjunto $\trcl(x)=\pred_{\in^*}(x)$, em que $\in^*$ é a transitivização de $\in$.
\end{definition}

A definição acima faz sentido pela Proposição~\ref{prop:fechoTransitivoDeRelacaoSetLike}, uma vez que $\in$ é set-like.

Note que classes transitivas são sinônimos de classes $\in$-fechadas.
\begin{proposition}[$\ZFmP$]\label{prop:propriedadesFechoTransitivo}
    Para quaisquer conjuntos $x$ e $y$:
    \begin{enumerate}[label=(\alph*)]
        \item $\trcl(x)$ é transitivo;
        \item $x\subseteq \trcl(x)$;
        \item se $T$ é uma classe transitiva e $x\subseteq T$, então $\trcl(x)\subseteq T$;
        \item se $T$ é uma classe transitiva e $x\in T$, então $\trcl(x)\subseteq T$;
        \item se $y\subseteq x$, então $\trcl(y)\subseteq \trcl(x)$;
        \item se $y\in^* x$, então $\trcl(y)\subseteq \trcl(x)$;
        \item $\trcl(x)=\bigcup_{y \in x}\bigl(\trcl(y)\cup\{y\}\bigr)$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    Tudo decorre imediatamente do Lema~\ref{lem:predRStarFechado} aplicado à relação $\in$.
\end{proof}

\begin{definition}[$\ZFmP$]
    Um conjunto $x$ é \textit{bem-fundado} se $\in$ é bem-fundada em $\trcl(x)$.

    A classe dos conjuntos bem-fundados é denotada por $\WF$.
\end{definition}

Alertamos o leitor para tomar cuidado para não se confundir com a nomenclatura:
$\in$ ser bem-fundada em $x$ não é o mesmo que $x$ ser bem-fundado.

O lema abaixo dá uma visão alternativa sobre o conceito de conjunto bem-fundado.

\begin{lemma}[$\ZFmP$]
    Um conjunto $X$ é bem-fundado se, e somente se, $\in$ é bem-fundada em alguma classe transitiva $t$ tal que $X\subseteq t$.
\end{lemma}
\begin{proof}
    Se $X$ é bem-fundado, então $\in$ é bem-fundada em $\trcl(X)$, que é transitiva e contém $X$.

    Já se $\in$ é bem-fundada em alguma classe transitiva $t$ que contém $X$, então $\in$ é bem-fundada em $\trcl(X)$, pois $\trcl(X)\subseteq t$.
\end{proof}

Conjuntos bem-fundados não formam $\in$-ciclos, como formaliza o próximo lema.

\begin{lemma}[$\ZFmP$]\label{lem:conjuntosBemFundadosNaoFazemCiclos}
    Se $x$ é um conjunto bem-fundado, então $x\notin^* x$.
\end{lemma}
\begin{proof}
    Suponha por absurdo que $x\in^* x$.
    Então $x \in \trcl(x)$.
    Mas $\{x\}\subseteq \trcl(x)$ é um conjunto não vazio sem elemento $\in^*$-minimal, contradizendo a Proposição~\ref{prop:fechoTransitivoBemFundadoEmA}.
\end{proof}

\begin{lemma}[$\ZFmP$]
    Seja $x$ um conjunto.
    Então $x$ é bem-fundado se, e somente se, $\in$ é bem-fundada em $\trcl(x)\cup\{x\}$.
\end{lemma}
\begin{proof}
    ($\Leftarrow$) Imediato, pois $\trcl(x)\subseteq \trcl(x)\cup\{x\}$.

    ($\Rightarrow$) Seja $X\subseteq \trcl(x)\cup\{x\}$ tal que $X\neq\emptyset$.
    Se $x\notin X$, então $X\subseteq \trcl(x)$, e, portanto, existe um elemento $\in$-minimal de $X$.

    Se $x\in X$, seja $Y=X\setminus\{x\}$.
    Se $Y=\emptyset$, então $X=\{x\}$, e $x$ é um elemento $\in$-minimal de $X$ pelo Lema~\ref{lem:conjuntosBemFundadosNaoFazemCiclos}.

    Se $Y\neq\emptyset$, então $Y\subseteq \trcl(x)$, e, portanto, existe um elemento $\in$-minimal $m$ de $Y$.
    Assim, todo $y\in Y$ é tal que $y\notin m$.
    Devemos ter também que $x\notin m$, do contrário teremos $x\in m\in^* x$, o que é absurdo pelo Lema~\ref{lem:conjuntosBemFundadosNaoFazemCiclos}.
    Logo, $m$ é um elemento $\in$-minimal de $X$.
\end{proof}

Agora iremos introduzir o conceito de rank de um conjunto bem-fundado.

\begin{definition}[$\ZFmP$]
    Se $x \in \WF$, então $\rank(x)=\rank_{\in, \trcl(x)\cup\{x\}}(x)$.
\end{definition}

\begin{proposition}[$\ZFmP$]\label{prop:rankRestricaoPertence}
    Se $T$ é uma classe transitiva tal que $\in$ é bem-fundada em $T$, então:
    \begin{enumerate}[label=(\alph*)]
        \item $T\subseteq \WF$.
        \item $\rank_{\in, T}(x)=\rank(x)$ para todo $x\in T$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    (a) Para cada $x\in T$, temos que $\trcl(x)\cup\{x\}\subseteq T$, pois $T$ é transitivo.
    Como $\in$ é bem-fundada em $T$, então $\in$ é bem-fundada em $\trcl(x)\cup\{x\}$, e, portanto, $x\in \WF$.

    (b) Como $T$ é transitivo, dado $x \in T$, temos que $\trcl(x)\cup\{x\}\subseteq T$.
    Seja $t=\trcl(x)\cup\{x\}$.
    Temos que $(\in_{T})_t=\in_t$.
    Temos ainda que $\pred_{(\in_T)^*}(x)\subseteq\pred_{\in^*}(x)=\trcl(x)\subseteq t$.
    
    Pela Proposição~\ref{prop:rankRestricao}, $\rank(x)=\rank_{\in, t}(x)=\rank_{\in_T}(x)=\rank_{\in, T}(x)$.
\end{proof}


\begin{proposition}[$\ZFmP$]
    $\WF$ é uma classe transitiva e $\in$ é bem-fundada em $\WF$.
\end{proposition}
\begin{proof}
    $\WF$ é transitiva, pois, se $x\in \WF$ e $y\in x$, então $y\in \WF$ já que $\in_{\trcl(y)}\subseteq \in_{\trcl(x)}$.

    A relação $\in$ é bem-fundada em $\WF$: dado $X\subseteq \WF$ tal que $X\neq\emptyset$, pelo Esquema da Substituição, $\{\rank(z):z\in X\}$ é um conjunto não vazio de ordinais.
    Tome $x\in X$ tal que $\rank(x)$ seja mínimo.
    Se $y\in x$, temos que $y\notin X$, pois $\rank(y)=\rank_{\in, \trcl(y)\cup\{y\}}(y)=\rank_{\in, \trcl(x)\cup\{x\}}(y)<\rank(x)$ pela Proposição~\ref{prop:rankRestricaoPertence}.
    Assim, $x$ é um elemento $\in$-minimal de $X$.
\end{proof}

Assim, para todo $x\in \WF$, $\rank(x)=\rank_{\in,\WF}(x)$ pela Proposição~\ref{prop:rankRestricaoPertence}.
\begin{corollary}[$\ZFmP$]
    Se $x\in \WF$, então $\rank(x)=\sup\{\rank(y)+1: y\in x\}$.
\end{corollary}
\begin{proof}
    Seja $x\in \WF$.
    Temos que $\rank(x)=\rank_{\in, \WF}(x)=\sup\{\rank(y)+1: y\in_{\WF} x\}=\sup\{\rank(y)+1: y\in x\}$.
\end{proof}
\begin{corollary}[$\ZFmP$]
    Se $x, y \in \WF$ e $x\in^* y$, então $\rank(x)<\rank(y)$.
\end{corollary}
\begin{proof}
    Decorre diretamente do Lema~\ref{lem:propriedadesRank}.
\end{proof}

\begin{corollary}[$\ZFmP$]
    Se $x, y \in \WF$ e $x\subseteq y$, então $\rank(x)\leq \rank(y)$.
\end{corollary}
\begin{proof}
    Temos que $\rank(x)=\sup\{\rank(z)+1: z\in x\}\leq \sup\{\rank(z)+1: z\in y\}=\rank(y)$.
\end{proof}
\begin{corollary}[$\ZFmP$]
    Para $x$ qualquer, temos que $x\in \WF$ se, e somente se, $x\subseteq \WF$.
\end{corollary}
\begin{proof}
    Se $x\in \WF$, então $x\subseteq \WF$ pois $\WF$ é transitiva.

    Se $x\subseteq \WF$, então $\trcl(x)\subseteq \WF$, pois $\WF$ é transitiva.
    Como $\WF$ é bem-fundada, então $\in$ é bem-fundada em $\trcl(x)$, e, portanto, $x\in \WF$.
\end{proof}

\begin{lemma}[$\ZFmP$]
    Para todo ordinal $\alpha$, $\alpha$ é bem-fundado e $\rank(\alpha)=\alpha$.
\end{lemma}
\begin{proof}
    Utilizamos indução transfinita.

    Suponha que para todo $\beta<\alpha$, $\beta \in \WF$ e $\rank(\beta)=\beta$.
    Então $\alpha=\{\beta: \beta<\alpha\}\subseteq \WF$, e, portanto, $\alpha\in \WF$.
    Ademais, $\rank(\alpha)=\sup\{\rank(\beta)+1: \beta\in \alpha\}=\sup\{\beta+1: \beta< \alpha\}=\alpha$.

    Para a última igualdade, com efeito, se $\beta<\alpha$, então $\beta+1\leq \alpha$, logo, $\alpha$ é um majorante de $A=\{\beta+1: \beta< \alpha\}$.
    Esse é o menor majorante, pois se $\xi<\alpha$, então $\xi+1\in A$, e, portanto, $\xi$ não é um majorante de $A$.
\end{proof}

Assim, $\ON\subseteq \WF$.

\begin{corollary}[$\ZFmP$]
    Sejam $x, y\in \WF$.
    Então:
    \begin{enumerate}[label=(\alph*)]
        \item $\rank(\{x, y\})=\max\{\rank(x), \rank(y)\}+1$.
        \item $\rank(\langle x, y\rangle)=\max\{\rank(x), \rank(y)\}+2$.
        \item $\bigcup x\in \WF$ e $\rank(\bigcup x)\leq \rank(x)$.
        \item $x\cup y$ é bem-fundado e $\rank(x\cup y)=\max\{\rank(x), \rank(y)\}$.
        \item $\trcl(x)\in \WF$ e $\rank(\trcl(x))=\rank(x)$.
        \item Se existe $\mathcal P(x)$, então $\mathcal P(x)\in \WF$ e $\rank(\mathcal P(x))=\rank(x)+1$.
    \end{enumerate}
\end{corollary}

\begin{proof}
    (a) Temos que $\{x, y\}\subseteq \WF$, logo, $\{x, y\}\in \WF$.

    Ademais, $\rank(\{x, y\})=\sup\{\rank(z)+1: z\in \{x, y\}\}=\max\{\rank(x)+1, \rank(y)+1\}=\max\{\rank(x), \rank(y)\}+1$.

    (b) Como $\langle x, y\rangle=\{\{x\}, \{x, y\}\}$, temos que $\langle x, y\rangle\in \WF$ pelo item (a).
    Ademais, $\rank(\langle x, y\rangle)=\sup\{\rank(z)+1: z\in \langle x, y\rangle\}=\sup\{\rank(\{x\})+1, \rank(\{x, y\})+1\}=\sup\{\rank(x)+2, \max\{\rank(x), \rank(y)\}+2\}=\max\{\rank(x), \rank(y)\}+2$.

    (c)
\end{proof}



\subsection*{Exercícios}
\begin{exercise}[$\ZFmP$]
    Seja $x$ um conjunto.
    Para cada $n$ natural, definimos $\bigcup^n x$ recursivamente de modo que:
    \begin{align*}
        \bigcup^0 x&=x\\
        \bigcup^{n+1} x&=\bigcup(\bigcup^n x).
    \end{align*}    

    Mostre que $\trcl(x)=\bigcup_{n\in\omega}\bigcup^n x$.
\end{exercise}
